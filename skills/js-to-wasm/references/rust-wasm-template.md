# Rust WASM Module Template

Scaffold for a new Rust-based WASM module in BentoPDF.

## Directory Structure

```
wasm/bentopdf-{feature}-wasm/
├── Cargo.toml
├── src/
│   └── lib.rs
├── tests/
│   └── integration.rs
├── build.sh
└── pkg/                  # Generated by wasm-pack (gitignored)
```

## src/lib.rs — Starter Template

```rust
use wasm_bindgen::prelude::*;
use serde::Deserialize;

/// Settings passed from JS. Fields match the TypeScript interface.
#[derive(Deserialize, Default)]
pub struct ProcessingSettings {
    // Add fields matching your JS settings object
    pub intensity: f32,
    pub enabled: bool,
}

/// Main processing function exposed to JS.
///
/// Accepts a pixel buffer (RGBA, 4 bytes per pixel) and settings as a JsValue.
/// Returns the processed pixel buffer.
#[wasm_bindgen]
pub fn process_pixels(pixels: &[u8], settings_js: JsValue) -> Result<Vec<u8>, JsError> {
    let settings: ProcessingSettings = serde_wasm_bindgen::from_value(settings_js)
        .map_err(|e| JsError::new(&format!("Invalid settings: {}", e)))?;

    let mut output = pixels.to_vec();
    let len = output.len();

    // Process in RGBA chunks
    let mut i = 0;
    while i + 3 < len {
        let r = output[i] as f32;
        let g = output[i + 1] as f32;
        let b = output[i + 2] as f32;
        // Alpha at output[i + 3] — usually passed through unchanged

        // --- Your processing logic here ---
        let (r_out, g_out, b_out) = process_pixel(r, g, b, &settings);

        output[i] = clamp_u8(r_out);
        output[i + 1] = clamp_u8(g_out);
        output[i + 2] = clamp_u8(b_out);
        // output[i + 3] unchanged (alpha)

        i += 4;
    }

    Ok(output)
}

/// Process a single pixel. Keep this as a separate function for testability.
#[inline]
fn process_pixel(r: f32, g: f32, b: f32, settings: &ProcessingSettings) -> (f32, f32, f32) {
    if !settings.enabled {
        return (r, g, b);
    }
    // Example: simple brightness adjustment
    let adj = settings.intensity * 2.55;
    (r + adj, g + adj, b + adj)
}

/// Clamp a float to 0..255 and convert to u8.
#[inline]
fn clamp_u8(v: f32) -> u8 {
    v.max(0.0).min(255.0) as u8
}

// ─── Color space utilities ───────────────────────────────────────

/// RGB to HSL conversion. Input: 0-255 per channel. Output: h in 0..1, s in 0..1, l in 0..1.
pub fn rgb_to_hsl(r: f32, g: f32, b: f32) -> (f32, f32, f32) {
    let r = r / 255.0;
    let g = g / 255.0;
    let b = b / 255.0;
    let max = r.max(g).max(b);
    let min = r.min(g).min(b);
    let l = (max + min) / 2.0;

    if (max - min).abs() < f32::EPSILON {
        return (0.0, 0.0, l);
    }

    let d = max - min;
    let s = if l > 0.5 { d / (2.0 - max - min) } else { d / (max + min) };
    let h = if (max - r).abs() < f32::EPSILON {
        ((g - b) / d + if g < b { 6.0 } else { 0.0 }) / 6.0
    } else if (max - g).abs() < f32::EPSILON {
        ((b - r) / d + 2.0) / 6.0
    } else {
        ((r - g) / d + 4.0) / 6.0
    };

    (h, s, l)
}

/// HSL to RGB conversion. Input: h, s, l all in 0..1. Output: 0-255 per channel.
pub fn hsl_to_rgb(h: f32, s: f32, l: f32) -> (f32, f32, f32) {
    if s.abs() < f32::EPSILON {
        let v = (l * 255.0).round();
        return (v, v, v);
    }

    let hue2rgb = |p: f32, q: f32, mut t: f32| -> f32 {
        if t < 0.0 { t += 1.0; }
        if t > 1.0 { t -= 1.0; }
        if t < 1.0 / 6.0 { return p + (q - p) * 6.0 * t; }
        if t < 0.5 { return q; }
        if t < 2.0 / 3.0 { return p + (q - p) * (2.0 / 3.0 - t) * 6.0; }
        p
    };

    let q = if l < 0.5 { l * (1.0 + s) } else { l + s - l * s };
    let p = 2.0 * l - q;

    (
        (hue2rgb(p, q, h + 1.0 / 3.0) * 255.0).round(),
        (hue2rgb(p, q, h) * 255.0).round(),
        (hue2rgb(p, q, h - 1.0 / 3.0) * 255.0).round(),
    )
}

// ─── Tests ───────────────────────────────────────────────────────

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_clamp_u8() {
        assert_eq!(clamp_u8(-10.0), 0);
        assert_eq!(clamp_u8(128.5), 128);
        assert_eq!(clamp_u8(300.0), 255);
    }

    #[test]
    fn test_rgb_hsl_roundtrip() {
        let (h, s, l) = rgb_to_hsl(100.0, 150.0, 200.0);
        let (r, g, b) = hsl_to_rgb(h, s, l);
        assert!((r - 100.0).abs() <= 1.0);
        assert!((g - 150.0).abs() <= 1.0);
        assert!((b - 200.0).abs() <= 1.0);
    }

    #[test]
    fn test_brightness_single_pixel() {
        let pixels = vec![100, 100, 100, 255];
        let settings = ProcessingSettings { intensity: 50.0, enabled: true };
        let settings_js = serde_wasm_bindgen::to_value(&settings).unwrap();
        // For native tests, call process_pixel directly instead
        let (r, g, b) = process_pixel(100.0, 100.0, 100.0, &settings);
        assert_eq!(clamp_u8(r), 227); // 100 + 50*2.55 = 227.5 → 227
    }
}
```

## build.sh

```bash
#!/bin/bash
set -e

# Build for web target (ES module output)
wasm-pack build --target web --release

# Optional: optimize WASM binary further with wasm-opt (binaryen)
if command -v wasm-opt &> /dev/null; then
  wasm-opt -Oz pkg/*_bg.wasm -o pkg/*_bg.wasm
  echo "wasm-opt applied"
fi

echo "Build complete. Output in pkg/"
echo "Files:"
ls -lh pkg/
```

## Development Workflow

```bash
# 1. Write Rust code
# 2. Run native tests (fast)
cargo test

# 3. Build WASM
wasm-pack build --target web --release

# 4. Serve locally for dev testing
#    In vite.config.ts, add a dev middleware to serve from wasm/{module}/pkg/
#    Or use a simple static file server

# 5. Run browser tests
wasm-pack test --headless --chrome

# 6. When ready to publish
npm publish pkg/ --access public
```

## Size Budget

Target <500KB for the `.wasm` binary after `wasm-opt -Oz`. For reference:
- Simple pixel processing: ~50-100KB
- Image codec: ~200-400KB
- Full PDF manipulation: ~1-5MB (like PyMuPDF/Ghostscript)

Use `opt-level = "z"` and `lto = true` in `Cargo.toml` `[profile.release]` to minimize size.
